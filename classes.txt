class Aeroflot:
    def __init__(self, name_of_arrival, type_of_plane):
        self.name_of_arrival = name_of_arrival
        self.type_of_plane = type_of_plane

    def numbers_of_voyager(self, number_of_voyage):
        self.number_of_voyage = number_of_voyage

    def sort_voyagers(self):
        return ('Sorted voyagers : '
                + str(sorted(self.number_of_voyage)))

    def print_all(self, input_place):
        if self.name_of_arrival == input_place:
            print('Voyagers %s and type of plane %s, '
                  'fly out to %s' % (self.number_of_voyage, self.type_of_plane, self.name_of_arrival))
