def generator(s):
    i = 0
    while i < len(s):
        yield i
        i += 1


if __name__ == '__main__':
    s = input('Input the string: ')
    for x in generator(s):
        if s[x].islower():
            print(s[x].upper(), end='')
        else:
            print(s[x].lower(), end='')
